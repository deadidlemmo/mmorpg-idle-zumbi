function toSafeNumber(value: unknown, fallback = 0) {
  const parsed = Number(value);

  return Number.isFinite(parsed) ? parsed : fallback;
}

function getOptionalNumber(value: unknown) {
  const parsed = Number(value);

  return Number.isFinite(parsed) ? parsed : undefined;
}

function getOptionalBoolean(value: unknown) {
  return typeof value === 'boolean' ? value : undefined;
}

function clampPercent(value: unknown) {
  const parsed = toSafeNumber(value, 0);

  return Math.max(0, Math.min(100, parsed));
}

function clampNumber(value: unknown, min: number, max: number) {
  const parsed = toSafeNumber(value, min);

  return Math.max(min, Math.min(parsed, max));
}

function formatSeconds(seconds?: number | null) {
  if (seconds === null || seconds === undefined) return '—';

  const safeSeconds = Math.max(0, Math.floor(seconds));
  const hours = Math.floor(safeSeconds / 3600);
  const minutes = Math.floor((safeSeconds % 3600) / 60);
  const remainingSeconds = safeSeconds % 60;

  if (hours > 0) {
    return `${hours}h ${minutes}min`;
  }

  if (minutes > 0) {
    return `${minutes}min ${remainingSeconds}s`;
  }

  return `${remainingSeconds}s`;
}

function getApiErrorMessage(error: unknown, fallback: string) {
  const apiError = error as {
    response?: {
      data?: {
        message?: string | string[];
      };
    };
  };

  const message = apiError.response?.data?.message;

  if (Array.isArray(message)) {
    return message.join(' ');
  }

  return message ?? fallback;
}

function normalizeStats(value: unknown): DashboardStats | null {
  if (!value || typeof value !== 'object') return null;

  const stats = value as LooseStatsSource;

  return {
    strength: toSafeNumber(stats.strength, 0),
    vitality: toSafeNumber(stats.vitality, 0),
    agility: toSafeNumber(stats.agility, 0),
    precision: toSafeNumber(stats.precision, 0),
    technique: toSafeNumber(stats.technique, 0),
    willpower: toSafeNumber(stats.willpower, 0),
  };
}

function getStatsScore(stats: DashboardStats) {
  return (
    stats.strength +
    stats.vitality +
    stats.agility +
    stats.precision +
    stats.technique +
    stats.willpower
  );
}

function resolveCharacterStats(
  overview: CharacterOverviewResponse | null,
  character: CharacterViewModelWithLayoutFields | null,
): DashboardStats {
  type LooseOverview = CharacterOverviewResponse & {
    totalStats?: DashboardStats;
    stats?: DashboardStats;
    characterStats?: DashboardStats;
  };

  type LooseCharacter = CharacterOverviewResponse['character'] & {
    totalStats?: DashboardStats;
    primaryStats?: DashboardStats;
    stats?: DashboardStats;
    baseStats?: DashboardStats;
  };

  const looseOverview = overview as LooseOverview | null;
  const looseCharacter = overview?.character as LooseCharacter | undefined;

  const candidates = [
    character?.totalStats,
    looseCharacter?.totalStats,
    looseOverview?.totalStats,
    looseOverview?.stats,
    looseOverview?.characterStats,
    looseCharacter?.primaryStats,
    looseCharacter?.stats,
    looseCharacter?.baseStats,
    character?.baseStats,
  ]
    .map(normalizeStats)
    .filter((stats): stats is DashboardStats => Boolean(stats));

  if (candidates.length <= 0) return EMPTY_STATS;

  return candidates.reduce((best, current) => {
    return getStatsScore(current) > getStatsScore(best) ? current : best;
  }, candidates[0]);
}

function flattenCombatSubMaps(
  maps: AutoCombatMapViewModel[],
  characterLevel: number,
) {
  return maps
    .flatMap((gameMap) => {
      return (gameMap.subMaps ?? []).map((subMap) => {
        return {
          ...subMap,
          map: {
            id: gameMap.id,
            name: gameMap.name,
            tier: gameMap.tier,
          },
          mapName: gameMap.name,
        };
      });
    })
    .filter((subMap) => {
      const minLevel = subMap.minLevel ?? 1;

      return characterLevel >= minLevel && getActiveEncounters(subMap).length > 0;
    })
    .sort((a, b) => {
      return (
        (a.tier ?? 0) - (b.tier ?? 0) ||
        (a.minLevel ?? 0) - (b.minLevel ?? 0) ||
        a.name.localeCompare(b.name)
      );
    });
}

function getDefaultSubMapId(params: {
  maps: AutoCombatMapViewModel[];
  characterLevel: number;
  status: AutoCombatStatusResponse | null;
}) {
  const subMaps = flattenCombatSubMaps(params.maps, params.characterLevel);
  const sessionSubMapId = params.status?.subMap?.id;

  if (
    sessionSubMapId &&
    subMaps.some((subMap) => subMap.id === sessionSubMapId)
  ) {
    return sessionSubMapId;
  }

  return subMaps[0]?.id ?? '';
}

function calculateRoundProgress(params: {
  session: AutoCombatSessionApiViewModel | null;
  isActive: boolean;
  now: number;
}) {
  if (!params.isActive || !params.session?.lastProcessedAt) {
    return 0;
  }

  const roundDuration = Math.max(1, params.session.roundDurationSeconds ?? 5);
  const lastProcessedAt = new Date(params.session.lastProcessedAt).getTime();

  if (!Number.isFinite(lastProcessedAt)) {
    return 0;
  }

  const elapsedSeconds = Math.max(
    0,
    Math.floor((params.now - lastProcessedAt) / 1000),
  );

  return clampPercent(((elapsedSeconds % roundDuration) / roundDuration) * 100);
}

  return mobs[mobs.length - 1];
}

function getThreatWeightPercent(
  encounter: AutoCombatEncounterViewModel,
  encounters: AutoCombatEncounterViewModel[],
) {
  const totalWeight = encounters.reduce((total, currentEncounter) => {
    return total + Math.max(0, currentEncounter.weight ?? 0);
  }, 0);

  if (totalWeight <= 0) return 0;

  return Math.round(((encounter.weight ?? 0) / totalWeight) * 100);
}

function getRealtimeEventDelay(event?: AutoCombatRealtimeEvent) {
  if (!event?.type) return 160;

  if (event.type === 'MOB_SPAWNED') return 180;

  if (event.type === 'MOB_DEFEATED') {
    return 320;
  }

  if (event.type === 'PLAYER_DEFEATED') {
    return 420;
  }

  if (event.type === 'POTION_USED') return 220;

  return 160;
}

function buildSessionTotalsFromStatus(
  status: AutoCombatStatusResponse | null,
  session: AutoCombatSessionApiViewModel | null,
): RealtimeSessionTotalsState | null {
  if (!status && !session) return null;

  const totalKills =
    status?.sessionSummary?.mobs?.totalKills ??
    status?.rewards?.mobs?.reduce((total, mob) => {
      return total + mob.kills;
    }, 0) ??
    0;

  const totalLoot =
    status?.sessionSummary?.loot?.totalQuantity ??
    status?.rewards?.loots?.reduce((total, loot) => {
      return total + loot.quantity;
    }, 0) ??
    0;

  return {
    sessionId: session?.id ?? null,

    totalCombats:
      status?.sessionSummary?.combat?.totalCombats ??
      session?.totalCombatsResolved ??
      status?.processing?.combatsResolved ??
      totalKills ??
      0,

    totalRounds:
      status?.sessionSummary?.combat?.totalRounds ??
      session?.totalRoundsResolved ??
      status?.processing?.roundsResolved ??
      0,

    totalKills,

    totalXpGained:
      status?.sessionSummary?.progression?.totalXpGained ??
      session?.totalXpGained ??
      status?.processing?.xpGained ??
      0,

    totalLoot,

    potionsUsed: status?.processing?.potionsUsed ?? 0,

    updatedAt: Date.now(),
  };
}

function buildProgressFromSource(
  source: CharacterProgressSource | null | undefined,
  sessionId?: string | null,
): RealtimeCharacterProgressState | null {
  if (!source) return null;

  const levelProgress = source.levelProgress ?? null;

  const totalXp =
    getOptionalNumber(source.totalXp) ??
    getOptionalNumber(levelProgress?.totalXp) ??
    getOptionalNumber(source.xp) ??
    getOptionalNumber(levelProgress?.xp);

  const level =
    getOptionalNumber(source.level) ?? getOptionalNumber(levelProgress?.level);

  if (totalXp === undefined && level === undefined) {
    return null;
  }

  return {
    sessionId: sessionId ?? null,

    level: level !== undefined ? Math.max(1, Math.floor(level)) : undefined,

    xp: totalXp !== undefined ? Math.max(0, Math.floor(totalXp)) : undefined,

    currentLevelXp:
      getOptionalNumber(source.currentLevelXp) ??
      getOptionalNumber(source.xpIntoCurrentLevel) ??
      getOptionalNumber(levelProgress?.currentLevelXp) ??
      getOptionalNumber(levelProgress?.xpIntoCurrentLevel),

    xpToNextLevel:
      getOptionalNumber(source.xpToNextLevel) ??
      getOptionalNumber(source.nextLevelXp) ??
      getOptionalNumber(levelProgress?.xpToNextLevel) ??
      getOptionalNumber(levelProgress?.nextLevelXp),

    nextLevelXp:
      getOptionalNumber(source.nextLevelXp) ??
      getOptionalNumber(source.xpToNextLevel) ??
      getOptionalNumber(levelProgress?.nextLevelXp) ??
      getOptionalNumber(levelProgress?.xpToNextLevel),

    xpProgressPercent:
      getOptionalNumber(source.xpProgressPercent) ??
      getOptionalNumber(levelProgress?.xpProgressPercent) ??
      getOptionalNumber(levelProgress?.progressPercent),

    xpIntoCurrentLevel:
      getOptionalNumber(source.xpIntoCurrentLevel) ??
      getOptionalNumber(source.currentLevelXp) ??
      getOptionalNumber(levelProgress?.xpIntoCurrentLevel) ??
      getOptionalNumber(levelProgress?.currentLevelXp),

    xpNeededForNextLevel:
      getOptionalNumber(source.xpNeededForNextLevel) ??
      getOptionalNumber(levelProgress?.xpNeededForNextLevel),

    currentLevelStartXp:
      getOptionalNumber(source.currentLevelStartXp) ??
      getOptionalNumber(levelProgress?.currentLevelStartXp),

    nextLevelRequiredXp:
      getOptionalNumber(source.nextLevelRequiredXp) ??

    isAtLevelCap:
      getOptionalBoolean(source.isAtLevelCap) ??
      getOptionalBoolean(levelProgress?.isAtLevelCap),

    updatedAt: Date.now(),
  };
}

function mergeProgressKeepingHighestXp(
  current: RealtimeCharacterProgressState | null,
  incoming: RealtimeCharacterProgressState | null,
): RealtimeCharacterProgressState | null {
  if (!incoming) return current;
  if (!current) return incoming;

  const currentXp = current.xp;
  const incomingXp = incoming.xp;

  if (
    currentXp !== undefined &&
    incomingXp !== undefined &&
    currentXp > incomingXp
  ) {
    return {
      ...incoming,
      ...current,
      sessionId: incoming.sessionId ?? current.sessionId ?? null,
      updatedAt: Date.now(),
    };
  }

  return {
    ...current,
    ...incoming,
    sessionId: incoming.sessionId ?? current.sessionId ?? null,
    level: incoming.level ?? current.level,
    xp: incoming.xp ?? current.xp,
    currentLevelXp: incoming.currentLevelXp ?? current.currentLevelXp,
    xpToNextLevel: incoming.xpToNextLevel ?? current.xpToNextLevel,
    nextLevelXp: incoming.nextLevelXp ?? current.nextLevelXp,
    xpProgressPercent:
      incoming.xpProgressPercent ?? current.xpProgressPercent,
    xpIntoCurrentLevel:
      incoming.xpIntoCurrentLevel ?? current.xpIntoCurrentLevel,
    xpNeededForNextLevel:
      incoming.xpNeededForNextLevel ?? current.xpNeededForNextLevel,
    currentLevelStartXp:
      incoming.currentLevelStartXp ?? current.currentLevelStartXp,
    nextLevelRequiredXp:
      incoming.nextLevelRequiredXp ?? current.nextLevelRequiredXp,
    isAtLevelCap: incoming.isAtLevelCap ?? current.isAtLevelCap,
    xpGained: incoming.xpGained ?? current.xpGained,
    leveledUp: incoming.leveledUp ?? current.leveledUp,
    levelsGained: incoming.levelsGained ?? current.levelsGained,
    updatedAt: Date.now(),
  };
}

function pickHighestProgress(
  ...progresses: Array<RealtimeCharacterProgressState | null>
) {
  return progresses.reduce<RealtimeCharacterProgressState | null>(
    (best, current) => mergeProgressKeepingHighestXp(best, current),
    null,
  );
}

  const [realtimeSessionTotals, setRealtimeSessionTotals] =
    useState<RealtimeSessionTotalsState | null>(null);
  const [realtimeEventQueue, setRealtimeEventQueue] = useState<
    AutoCombatRealtimeEvent[]
  >([]);

  const [isPreviewLoading, setIsPreviewLoading] = useState(false);
  const [isLoading, setIsLoading] = useState(true);
  const [isActionLoading, setIsActionLoading] = useState(false);
  const [errorMessage, setErrorMessage] = useState('');
  const [nowTick, setNowTick] = useState(() => Date.now());

  const activeSessionIdRef = useRef<string | null>(null);
  const realtimeEventQueueLengthRef = useRef(0);
  const processedRealtimeEventKeysRef = useRef<Set<string>>(new Set());

  useEffect(() => {
    realtimeEventQueueLengthRef.current = realtimeEventQueue.length;
  }, [realtimeEventQueue.length]);

  const loadAutoCombatData = useCallback(async () => {
    if (!characterId) return;

    try {
      setErrorMessage('');

      const [overviewData, statusData, mapsData] = await Promise.all([
        getCharacterOverview(characterId),
        getAutoCombatStatus(characterId).catch(() => null),
        getAutoCombatMaps().catch(() => []),
      ]);

      const statusSession = getSessionFromStatus(statusData);
      const statusIsActive = isSessionActive(statusData, statusSession);

      const statusProgress = buildProgressFromStatus(statusData, statusSession);
      const overviewProgress = buildProgressFromSource(
        overviewData.character as CharacterProgressSource,
        statusSession?.id ?? null,
      );

      const mergedProgress = pickHighestProgress(
        overviewProgress,
        statusProgress,
      );

      setOverview(overviewData);
      setAutoCombatStatus(statusData);
      setMaps(mapsData);

      setRealtimeCharacterProgress((current) => {
        if (!statusIsActive) {
          return mergedProgress;
        }

        if (current && current.sessionId === statusSession?.id) {
          return current;
        }

        return current ?? overviewProgress ?? statusProgress ?? mergedProgress;
      });

      setRealtimeSessionTotals((current) => {
        if (!statusIsActive || !statusSession) {
          return current;
        }

        if (current?.sessionId === statusSession.id) {
          return current;
        }

        return buildSessionTotalsFromStatus(statusData, statusSession);
      });

      const characterLevel =
        mergedProgress?.level ?? overviewData.character.level ?? 1;

      setSelectedSubMapId((currentValue) => {
        const availableSubMaps = flattenCombatSubMaps(mapsData, characterLevel);

        if (
          currentValue &&
          availableSubMaps.some((subMap) => subMap.id === currentValue)
        ) {
          return currentValue;
        }

        return getDefaultSubMapId({
          maps: mapsData,
          characterLevel,
          status: statusData,
        });
      });
    } catch (error) {
      setErrorMessage(
        getApiErrorMessage(
          error,
          'Não foi possível carregar os dados do combate automático.',
        ),
      );
    }
  }, [characterId]);

  const applyRealtimeEvent = useCallback((payload: AutoCombatRealtimeEvent) => {
    const eventType = payload.type ?? '';
    const progressPayload = payload as AutoCombatRealtimeEventWithProgress;

    if (eventType === 'MOB_DEFEATED') {
      const hasProgressPayload =
        progressPayload.xpGained !== undefined ||
        progressPayload.characterXp !== undefined ||
        progressPayload.characterLevel !== undefined ||
        progressPayload.currentLevelXp !== undefined ||
        progressPayload.xpToNextLevel !== undefined ||
        progressPayload.xpProgressPercent !== undefined ||
        progressPayload.leveledUp !== undefined;

      if (hasProgressPayload) {
        setRealtimeCharacterProgress((current) => {
          const xpGained = getOptionalNumber(progressPayload.xpGained);

          const nextXp =
            getOptionalNumber(progressPayload.characterXp) ??
            (current?.xp !== undefined && xpGained !== undefined
              ? current.xp + xpGained
              : current?.xp);

          const nextLevel =
            getOptionalNumber(progressPayload.characterLevel) ??
            current?.level;

          const incoming: RealtimeCharacterProgressState = {
            sessionId: progressPayload.sessionId ?? current?.sessionId ?? null,

            level:
              nextLevel !== undefined
                ? Math.max(1, Math.floor(nextLevel))
                : current?.level,

            xp:
              nextXp !== undefined
                ? Math.max(0, Math.floor(nextXp))
                : current?.xp,

            currentLevelXp:
              getOptionalNumber(progressPayload.currentLevelXp) ??
              getOptionalNumber(progressPayload.xpIntoCurrentLevel) ??
              current?.currentLevelXp,

            xpToNextLevel:
              getOptionalNumber(progressPayload.xpToNextLevel) ??
              getOptionalNumber(progressPayload.nextLevelXp) ??
              current?.xpToNextLevel,

            nextLevelXp:
              getOptionalNumber(progressPayload.nextLevelXp) ??
              getOptionalNumber(progressPayload.xpToNextLevel) ??
              current?.nextLevelXp,

            xpProgressPercent:
              getOptionalNumber(progressPayload.xpProgressPercent) ??
              current?.xpProgressPercent,

            xpIntoCurrentLevel:
              getOptionalNumber(progressPayload.xpIntoCurrentLevel) ??
              getOptionalNumber(progressPayload.currentLevelXp) ??
              current?.xpIntoCurrentLevel,

            xpNeededForNextLevel:
              getOptionalNumber(progressPayload.xpNeededForNextLevel) ??
              current?.xpNeededForNextLevel,

            currentLevelStartXp:
              getOptionalNumber(progressPayload.currentLevelStartXp) ??
              current?.currentLevelStartXp,

            nextLevelRequiredXp:
              getOptionalNumber(progressPayload.nextLevelRequiredXp) ??
              current?.nextLevelRequiredXp,

            isAtLevelCap:
              getOptionalBoolean(progressPayload.isAtLevelCap) ??
              current?.isAtLevelCap,

            xpGained,
            leveledUp:
              getOptionalBoolean(progressPayload.leveledUp) ??
              current?.leveledUp,

            levelsGained:
              getOptionalNumber(progressPayload.levelsGained) ??
              current?.levelsGained,

            updatedAt: Date.now(),
          };

          return mergeProgressKeepingHighestXp(current, incoming);
        });
      }
    }

    const hasSessionTotalsPayload =
      payload.totalKills !== undefined ||
      payload.totalXpGained !== undefined ||
      payload.totalLoot !== undefined ||
      payload.potionsUsed !== undefined;

    if (hasSessionTotalsPayload) {
      setRealtimeSessionTotals((current) => {
        const totalKills =
          getOptionalNumber(payload.totalKills) ?? current?.totalKills;

        return {
          sessionId: payload.sessionId ?? current?.sessionId ?? null,

          totalCombats:
            totalKills ??
            getOptionalNumber(payload.combatIndex) ??
            current?.totalCombats,

          totalRounds: current?.totalRounds,

          totalKills,

          totalXpGained:
            getOptionalNumber(payload.totalXpGained) ?? current?.totalXpGained,

          totalLoot:
            getOptionalNumber(payload.totalLoot) ?? current?.totalLoot,

          potionsUsed:
            getOptionalNumber(payload.potionsUsed) ?? current?.potionsUsed,

          updatedAt: Date.now(),
        };
      });
    }

    setRealtimeCombat((current) => {
      const fallbackMobMaxHp =
        current?.mobMaxHp ?? payload.mobCurrentHp ?? payload.mobMaxHp ?? 0;

      const fallbackCharacterMaxHp =
        current?.characterMaxHp ??
        payload.characterCurrentHp ??
        payload.characterMaxHp ??
        0;

      const mobMaxHp = Math.max(
        0,
        toSafeNumber(payload.mobMaxHp, fallbackMobMaxHp),
      );

      const characterMaxHp = Math.max(
        0,
        toSafeNumber(payload.characterMaxHp, fallbackCharacterMaxHp),
      );

      const receivedMobHp = toSafeNumber(
        payload.mobCurrentHp,
        current?.mobCurrentHp ?? mobMaxHp,
      );

      const receivedCharacterHp = toSafeNumber(
        payload.characterCurrentHp,
        current?.characterCurrentHp ?? characterMaxHp,
      );

      const mobCurrentHp =
        eventType === 'MOB_DEFEATED'
          ? 0
          : clampNumber(receivedMobHp, 0, Math.max(0, mobMaxHp));

      const characterCurrentHp =
        eventType === 'PLAYER_DEFEATED'
          ? 0
          : clampNumber(receivedCharacterHp, 0, Math.max(0, characterMaxHp));

      return {
        ...current,

        mobId: payload.mobId ?? current?.mobId,
        mobName: payload.mobName ?? current?.mobName,
        mobCurrentHp,
        mobMaxHp,
        mobHpPercent:
          payload.mobHpPercent ?? clampPercent((mobCurrentHp / mobMaxHp) * 100),

        characterCurrentHp,
        characterMaxHp,
        characterHpPercent:
          payload.characterHpPercent ??
          clampPercent((characterCurrentHp / characterMaxHp) * 100),

        lastMessage: payload.message ?? current?.lastMessage,
        lastDamage: payload.damage ?? current?.lastDamage,
        lastEventType: payload.type ?? current?.lastEventType,
        isCritical: payload.isCritical ?? false,
        isDodged: payload.isDodged ?? false,

        actor: payload.actor ?? current?.actor,
        target: payload.target ?? current?.target,

        round: payload.round ?? current?.round,
        combatIndex: payload.combatIndex ?? current?.combatIndex,

        totalKills: payload.totalKills ?? current?.totalKills,
        totalXpGained: payload.totalXpGained ?? current?.totalXpGained,
        totalLoot: payload.totalLoot ?? current?.totalLoot,
        potionsUsed: payload.potionsUsed ?? current?.potionsUsed,

        updatedAt: Date.now(),
      };
    });

    setNowTick(Date.now());
  }, []);

  const enqueueRealtimeEvent = useCallback((payload: AutoCombatRealtimeEvent) => {
    const activeSessionId = activeSessionIdRef.current;

    if (
      payload.sessionId &&
      activeSessionId &&
      payload.sessionId !== activeSessionId
    ) {
      return;
    }

    const eventKey = getRealtimeEventKey(payload);

    if (processedRealtimeEventKeysRef.current.has(eventKey)) {
      return;
    }

    setRealtimeEventQueue((currentQueue) => {
      const alreadyQueued = currentQueue.some((queuedEvent) => {
        return getRealtimeEventKey(queuedEvent) === eventKey;
      });

      if (alreadyQueued) {
        return currentQueue;
      }

      const nextQueue = [...currentQueue, payload];

      return nextQueue.slice(-60);
    });
  }, []);

  const handleRealtimeStatusPayload = useCallback(
    (payload: AutoCombatStatusResponse) => {
      const payloadSession = getSessionFromStatus(payload);
      const payloadIsActive = isSessionActive(payload, payloadSession);
      const payloadIsTerminal = isTerminalSessionStatus(payloadSession?.status);
      const statusProgress = buildProgressFromStatus(payload, payloadSession);

      setAutoCombatStatus(payload);

      setRealtimeCharacterProgress((current) => {
        if (payloadIsActive) {
          if (!statusProgress) {
            return current;
          }

          if (!current || current.sessionId !== payloadSession?.id) {
            return statusProgress;
          }

          return current;
        }

        if (payloadIsTerminal && realtimeEventQueueLengthRef.current > 0) {
          return current;
        }

        return mergeProgressKeepingHighestXp(current, statusProgress);
      });

      setNowTick(Date.now());

      if (payloadSession) {
        setHasStartedHunt(true);
      }

      if (payloadIsActive && payloadSession) {
        setRealtimeSessionTotals((current) => {
          if (current?.sessionId === payloadSession.id) {
            return current;
          }

          return buildSessionTotalsFromStatus(payload, payloadSession);
        });

        setHasStartedHunt(true);
      }

      if (payloadIsTerminal) {
        setRealtimeSessionTotals((current) => {
          if (realtimeEventQueueLengthRef.current > 0) {
            return current ?? buildSessionTotalsFromStatus(payload, payloadSession);
          }

          return buildSessionTotalsFromStatus(payload, payloadSession) ?? current;
        });
      }
    },
    [],
  );

  const character = useMemo(() => {
    if (!overview) return null;

    return buildCharacterViewModel(overview);
  }, [overview]);

  const totalStats = useMemo(() => {
    return resolveCharacterStats(overview, character);
  }, [overview, character]);

  const session = useMemo(() => {
    return getSessionFromStatus(autoCombatStatus);
  }, [autoCombatStatus]);

  const hasActiveSession = isSessionActive(autoCombatStatus, session);
  const hasPendingRealtimeEvents = realtimeEventQueue.length > 0;
  const showActiveSession = hasActiveSession || hasPendingRealtimeEvents;

  const overviewCharacterProgress = useMemo(() => {
    return buildProgressFromSource(
      character as CharacterProgressSource | null | undefined,
      session?.id ?? null,
    );
  }, [character, session?.id]);

  const statusCharacterProgress = useMemo(() => {
    return buildProgressFromStatus(autoCombatStatus, session);
  }, [autoCombatStatus, session]);

  const visibleCharacterProgress = useMemo(() => {
    if (showActiveSession) {
      return (
        realtimeCharacterProgress ??
        statusCharacterProgress ??
        overviewCharacterProgress
      );
    }

    return pickHighestProgress(
      overviewCharacterProgress,
      statusCharacterProgress,
      realtimeCharacterProgress,
    );
  }, [
    overviewCharacterProgress,
    realtimeCharacterProgress,
    showActiveSession,
    statusCharacterProgress,
  ]);

  const { isConnected: isSocketConnected } = useAutoCombatSocket({
    characterId: characterId ?? null,
    enabled: Boolean(characterId),

    onStatus: handleRealtimeStatusPayload,
    onSessionUpdated: handleRealtimeStatusPayload,

    onFinished: (payload) => {
      handleRealtimeStatusPayload(payload);
    },

    onStopped: (payload) => {
      handleRealtimeStatusPayload(payload);
      setRealtimeEventQueue([]);
      setRealtimeCombat(null);
      setRealtimeSessionTotals(null);
      processedRealtimeEventKeysRef.current.clear();
    },

    onMobSpawned: (payload) => {
      enqueueRealtimeEvent(payload);
    },

    onHit: (payload) => {
      enqueueRealtimeEvent(payload);
    },

    onMobDefeated: (payload) => {
      enqueueRealtimeEvent(payload);
    },

    onPlayerDefeated: (payload) => {
      enqueueRealtimeEvent(payload);
    },

    onPotionUsed: (payload) => {
      enqueueRealtimeEvent(payload);
    },

    onError: (message) => {
      setErrorMessage(message);
    },
  });

  const firstQueuedRealtimeEvent = realtimeEventQueue[0] ?? null;

  useEffect(() => {
    if (!firstQueuedRealtimeEvent) return;

    const eventKey = getRealtimeEventKey(firstQueuedRealtimeEvent);

    processedRealtimeEventKeysRef.current.add(eventKey);

    if (processedRealtimeEventKeysRef.current.size > 500) {
      processedRealtimeEventKeysRef.current = new Set(
        Array.from(processedRealtimeEventKeysRef.current).slice(-250),
      );
    }

    applyRealtimeEvent(firstQueuedRealtimeEvent);

    const timeoutId = window.setTimeout(() => {
      setRealtimeEventQueue((currentQueue) => currentQueue.slice(1));
    }, getRealtimeEventDelay(firstQueuedRealtimeEvent));

    return () => {
      window.clearTimeout(timeoutId);
    };
  }, [firstQueuedRealtimeEvent, applyRealtimeEvent]);

  useEffect(() => {
    activeSessionIdRef.current = session?.id ?? null;
  }, [session?.id]);

  useEffect(() => {
    let isMounted = true;

    async function load() {
      if (!characterId) return;

      try {
        setIsLoading(true);
        await loadAutoCombatData();
      } finally {
        if (isMounted) {
          setIsLoading(false);
        }
      }
    }

    load();

    return () => {
      isMounted = false;
    };
  }, [characterId, loadAutoCombatData]);

  useEffect(() => {
    if (hasActiveSession) {
      setHasStartedHunt(true);
    }
  }, [hasActiveSession]);

  useEffect(() => {
    if (!hasActiveSession && realtimeEventQueue.length <= 0) {
      const timeoutId = window.setTimeout(() => {
        setRealtimeCombat(null);
      }, 500);

      return () => {
        window.clearTimeout(timeoutId);
      };
    }

    return undefined;
  }, [hasActiveSession, realtimeEventQueue.length]);

  useEffect(() => {
    if (!showActiveSession) return;

    const intervalId = window.setInterval(() => {
      setNowTick(Date.now());
    }, 1000);

    return () => window.clearInterval(intervalId);
  }, [showActiveSession]);

  useEffect(() => {
    if (!hasActiveSession || isSocketConnected) return;

    const intervalId = window.setInterval(() => {
      loadAutoCombatData();
    }, 5000);

    return () => window.clearInterval(intervalId);
  }, [hasActiveSession, isSocketConnected, loadAutoCombatData]);

  const availableSubMaps = useMemo(() => {
    return flattenCombatSubMaps(
      maps,
      visibleCharacterProgress?.level ?? character?.level ?? 1,
    );
  }, [maps, visibleCharacterProgress?.level, character?.level]);

  const selectedSubMap = useMemo(() => {
    return availableSubMaps.find((subMap) => subMap.id === selectedSubMapId);
  }, [availableSubMaps, selectedSubMapId]);

  const selectedMap = useMemo(() => {
    if (!selectedSubMap) return null;

    return maps.find((gameMap) => {
      return gameMap.subMaps?.some((subMap) => subMap.id === selectedSubMap.id);
    });
  }, [maps, selectedSubMap]);

  const selectedSubMapThreats = useMemo(() => {
    return getActiveEncounters(selectedSubMap).sort((a, b) => {
      return (b.weight ?? 0) - (a.weight ?? 0);
    });
  }, [selectedSubMap]);

  useEffect(() => {
    let isMounted = true;

    async function loadPreview() {
      if (!characterId || !selectedSubMapId || !hasStartedHunt) {
        setPreparationPreview(null);
        return;
      }

      if (showActiveSession) {
        setPreparationPreview(null);
        return;
      }

      try {
        setIsPreviewLoading(true);

        const data = await previewAutoCombat({
          characterId,
          subMapId: selectedSubMapId,
          projectionSeconds: 1800,
        });

        if (isMounted) {
          setPreparationPreview(data.combatPreview ?? null);
        }
      } catch {
        if (isMounted) {
          setPreparationPreview(null);
        }
      } finally {
        if (isMounted) {
          setIsPreviewLoading(false);
        }
      }
    }

    loadPreview();

    return () => {
      isMounted = false;
    };
  }, [characterId, selectedSubMapId, hasStartedHunt, showActiveSession]);

  if (!characterId) {
    return <Navigate to="/characters" replace />;
  }

  if (isLoading) {
    return (
      <main className="dashboard-loading">
        <div className="loading-spinner" />
        <span>Carregando combate automático...</span>
      </main>
    );
  }

  if (!character) {
    return <Navigate to="/characters" replace />;
  }

  const characterWithPotionConfig = character as CharacterWithSinglePotionConfig;

  const potionConfigs =
    characterWithPotionConfig.potionConfigs ??
    (characterWithPotionConfig.potionConfig
      ? [characterWithPotionConfig.potionConfig]
      : []);

  const potionSlots = Array.from({ length: 3 }, (_, index) => {
    return potionConfigs[index] ?? null;
  });

  const latestKilledMob = getLatestKilledMob(autoCombatStatus);
  const mainThreat = selectedSubMapThreats[0] ?? null;
  const remainingSeconds = getRemainingSeconds(autoCombatStatus);

  const rawCharacterMaxHp =
    showActiveSession && realtimeCombat?.characterMaxHp !== undefined
      ? realtimeCombat.characterMaxHp
      : hasActiveSession
        ? autoCombatStatus?.character?.maxHp ??
          autoCombatStatus?.sessionSummary?.hp?.max ??
          character.maxHp
        : character.maxHp;

  const currentCharacterMaxHp = Math.max(
    1,
    toSafeNumber(rawCharacterMaxHp, character.maxHp ?? 1),
  );

  const rawCharacterHp =
    showActiveSession && realtimeCombat?.characterCurrentHp !== undefined
      ? realtimeCombat.characterCurrentHp
      : hasActiveSession
        ? autoCombatStatus?.character?.currentHp ??
          autoCombatStatus?.sessionSummary?.hp?.current ??
          character.currentHp
        : character.currentHp;

  const currentCharacterHp = clampNumber(
    rawCharacterHp,
    0,
    currentCharacterMaxHp,
  );

  const currentCharacterLevel =
    visibleCharacterProgress?.level ??
    autoCombatStatus?.character?.level ??
    character.level ??
    1;

  const currentCharacterXp =
    visibleCharacterProgress?.xp ??
    (showActiveSession
      ? character.totalXp ?? character.xp ?? 0
      : autoCombatStatus?.character?.totalXp ??
        autoCombatStatus?.character?.levelProgress?.totalXp ??
        autoCombatStatus?.character?.xp ??
        character.totalXp ??
        character.xp ??
        0);

  const currentLevelXp =
    visibleCharacterProgress?.currentLevelXp ??
    visibleCharacterProgress?.xpIntoCurrentLevel ??
    character.currentLevelXp ??
    character.levelProgress?.currentLevelXp ??
    character.levelProgress?.xpIntoCurrentLevel ??
    undefined;

  const xpToNextLevel =
    visibleCharacterProgress?.xpToNextLevel ??
    visibleCharacterProgress?.nextLevelXp ??
    character.xpToNextLevel ??
    character.nextLevelXp ??
    character.levelProgress?.xpToNextLevel ??
    character.levelProgress?.nextLevelXp ??
    undefined;

  const xpProgressPercent =
    visibleCharacterProgress?.xpProgressPercent ??
    character.xpProgressPercent ??
    character.levelProgress?.xpProgressPercent ??
    character.levelProgress?.progressPercent ??
    undefined;

  const xpIntoCurrentLevel =
    visibleCharacterProgress?.xpIntoCurrentLevel ??
    visibleCharacterProgress?.currentLevelXp ??
    character.xpIntoCurrentLevel ??
    character.currentLevelXp ??
    character.levelProgress?.xpIntoCurrentLevel ??
    character.levelProgress?.currentLevelXp ??
    undefined;

  const xpNeededForNextLevel =
    visibleCharacterProgress?.xpNeededForNextLevel ??
    character.xpNeededForNextLevel ??
    character.levelProgress?.xpNeededForNextLevel ??
    undefined;

  const currentLevelStartXp =
    visibleCharacterProgress?.currentLevelStartXp ??
    character.currentLevelStartXp ??
    character.levelProgress?.currentLevelStartXp ??
    undefined;

  const nextLevelRequiredXp =
    visibleCharacterProgress?.nextLevelRequiredXp ??
    character.nextLevelRequiredXp ??
    character.levelProgress?.nextLevelRequiredXp ??
    undefined;

  const isAtLevelCap =
    visibleCharacterProgress?.isAtLevelCap ??
    character.isAtLevelCap ??
    character.levelProgress?.isAtLevelCap ??
    undefined;

  const activeSessionMapName = showActiveSession
    ? autoCombatStatus?.subMap?.map?.name
    : undefined;

  const currentLayoutMapName =
    activeSessionMapName ??
    selectedMap?.name ??
    selectedSubMap?.map?.name ??
    selectedSubMap?.mapName ??
    character.currentMapName;

  const layoutCharacter: CharacterViewModelWithLayoutFields = {
    ...character,
    level: currentCharacterLevel,
    xp: currentCharacterXp,
    totalXp: currentCharacterXp,

    currentLevelXp,
    xpToNextLevel,
    nextLevelXp: xpToNextLevel,
    xpProgressPercent,
    xpIntoCurrentLevel,
    xpNeededForNextLevel,
    currentLevelStartXp,
    nextLevelRequiredXp,
    isAtLevelCap,

    currentHp: currentCharacterHp,
    maxHp: currentCharacterMaxHp,
    currentMapName: currentLayoutMapName,
  };

  const characterHasHp = currentCharacterHp > 0;

  const characterHpPercent =
    currentCharacterMaxHp > 0
      ? (currentCharacterHp / currentCharacterMaxHp) * 100
      : 0;

  const characterHpStyle = {
    width: `${clampPercent(characterHpPercent)}%`,
  } as CSSProperties;

  const activeMobName =
    realtimeCombat?.mobName ??
    autoCombatStatus?.currentMob?.name ??
    mainThreat?.mob?.name ??
    latestKilledMob?.mobName ??
    'Aguardando mob';

  const rawActiveMobMaxHp =
    realtimeCombat?.mobMaxHp ??
    autoCombatStatus?.currentMob?.maxHp ??
    autoCombatStatus?.currentMob?.hp ??
    mainThreat?.mob?.hp ??
    0;

  const activeMobMaxHp = Math.max(0, toSafeNumber(rawActiveMobMaxHp, 0));

  const rawActiveMobCurrentHp =
    realtimeCombat?.mobCurrentHp ??
    autoCombatStatus?.currentMob?.currentHp ??
    (activeMobMaxHp > 0 ? activeMobMaxHp : 0);

  const activeMobCurrentHp = clampNumber(
    rawActiveMobCurrentHp,
    0,
    activeMobMaxHp,
  );

  const activeMobHpPercent =
    activeMobMaxHp > 0 ? (activeMobCurrentHp / activeMobMaxHp) * 100 : 0;

  const activeMobHpStyle = {
    width: `${clampPercent(activeMobHpPercent)}%`,
  } as CSSProperties;

  const activeMobReference = realtimeCombat?.combatIndex
    ? `Combate ${realtimeCombat.combatIndex}${
        realtimeCombat.round ? ` · Rodada ${realtimeCombat.round}` : ''
      }`
    : autoCombatStatus?.session?.currentCombatIndex
      ? `Combate ${autoCombatStatus.session.currentCombatIndex}${
          autoCombatStatus.session.currentRound
            ? ` · Rodada ${autoCombatStatus.session.currentRound}`
            : ''
        }`
      : mainThreat?.mob
        ? `Nv. ${mainThreat.mob.level}`
        : latestKilledMob
          ? `${latestKilledMob.kills} abate(s)`
          : '—';

  const roundProgressPercent = calculateRoundProgress({
    session,
    isActive: showActiveSession,
    now: nowTick,
  });

  const roundProgressStyle = {
    width: `${roundProgressPercent}%`,
  } as CSSProperties;

  const sessionStatusText =
    autoCombatStatus?.sessionSummary?.statusText ??
    formatSessionStatus(session?.status);

  const statusSessionTotals = buildSessionTotalsFromStatus(
    autoCombatStatus,
    session,
  );

  const visibleRealtimeSessionTotals =
    realtimeSessionTotals?.sessionId && session?.id
      ? realtimeSessionTotals.sessionId === session.id
        ? realtimeSessionTotals
        : null
      : realtimeSessionTotals;

  const visibleSessionTotals =
    showActiveSession && isSocketConnected
      ? visibleRealtimeSessionTotals ?? statusSessionTotals
      : statusSessionTotals ?? visibleRealtimeSessionTotals;

  const totalKills =
    visibleSessionTotals?.totalKills ??
    realtimeCombat?.totalKills ??
    autoCombatStatus?.sessionSummary?.mobs?.totalKills ??
    autoCombatStatus?.rewards?.mobs?.reduce((total, mob) => {
      return total + mob.kills;
    }, 0) ??
    0;

import { getAuthToken } from '../../../services/api/authToken';
import {

    type AutoCombatConnectedPayload,
    type AutoCombatJoinedPayload,
    type AutoCombatLeftPayload,
    type AutoCombatRealtimeEvent,
    type AutoCombatSocket,
    type AutoCombatSocketError,
} from '../../../services/websocket/socketClient';
import type { AutoCombatStatusResponse } from '../types/auto-combat.types';

type UseAutoCombatSocketOptions = {
  characterId?: string | null;
  enabled?: boolean;

  onStatus?: (payload: AutoCombatStatusResponse) => void;
  onSessionUpdated?: (payload: AutoCombatStatusResponse) => void;
  onFinished?: (payload: AutoCombatStatusResponse) => void;
  onStopped?: (payload: AutoCombatStatusResponse) => void;

  onMobSpawned?: (payload: AutoCombatRealtimeEvent) => void;
  onHit?: (payload: AutoCombatRealtimeEvent) => void;
  onMobDefeated?: (payload: AutoCombatRealtimeEvent) => void;
  onPlayerDefeated?: (payload: AutoCombatRealtimeEvent) => void;
  onPotionUsed?: (payload: AutoCombatRealtimeEvent) => void;

  onError?: (message: string) => void;
};

export function useAutoCombatSocket(options: UseAutoCombatSocketOptions) {
  const {
    characterId,
    enabled = true,
    onStatus,
    onSessionUpdated,
    onFinished,
    onStopped,
    onMobSpawned,
    onHit,
    onMobDefeated,
    onPlayerDefeated,
    onPotionUsed,
    onError,
  } = options;

  const [socket, setSocket] = useState<AutoCombatSocket | null>(null);
  const [isConnected, setIsConnected] = useState(false);
  const [isJoined, setIsJoined] = useState(false);
  const [errorMessage, setErrorMessage] = useState('');
  const [lastEvent, setLastEvent] = useState<AutoCombatRealtimeEvent | null>(
    null,
  );

    onStatus,
    onSessionUpdated,
    onFinished,
    onStopped,
    onMobSpawned,
    onHit,
    onMobDefeated,
    onPlayerDefeated,
    onPotionUsed,
    onError,
  });

    onStatus,
    onSessionUpdated,
    onFinished,
    onStopped,
    onMobSpawned,
    onHit,
    onMobDefeated,
    onPlayerDefeated,
    onPotionUsed,
    onError,
  };

    if (!enabled || !characterId) {
      setSocket(null);
      setIsConnected(false);
      setIsJoined(false);
      setLastEvent(null);
      return;
    }

    if (!token) {
      const message =
        'Token de autenticação não encontrado. Faça login novamente.';

      setSocket(null);
      setIsConnected(false);
      setIsJoined(false);
      setErrorMessage(message);
      handlersRef.current.onError?.(message);

    let hasJoinedCharacterRoom = false;
    let isJoiningCharacterRoom = false;
    let joinTimeoutId: number | undefined;

    setLastEvent(null);

      token,
    };

    function safeSetError(message: string) {
      if (isDisposed) return;

      setErrorMessage(message);
      handlersRef.current.onError?.(message);
    }

    function clearSocketError() {
      if (isDisposed) return;

      setErrorMessage('');
    }

    function joinCharacterRoom() {
      if (
        isDisposed ||
        !characterId ||
        !currentSocket.connected ||
        hasJoinedCharacterRoom ||
        isJoiningCharacterRoom
      ) {
        return;
      }

      isJoiningCharacterRoom = true;

      currentSocket.emit('auto-combat:join', {
        characterId,
      });
    }

    function scheduleJoinCharacterRoom() {
      if (joinTimeoutId !== undefined) {
        window.clearTimeout(joinTimeoutId);
      }

      joinTimeoutId = window.setTimeout(() => {
        joinCharacterRoom();
      }, 50);
    }

    function handleConnect() {
      if (isDisposed) return;

      setIsConnected(true);
      clearSocketError();

      /*
       * Não entramos imediatamente na sala aqui.
       * O backend ainda pode estar validando o JWT no handleConnection.
       * A entrada principal acontece no evento "auto-combat:connected".
       */
    }

    function handleAuthenticatedConnection(
      _payload: AutoCombatConnectedPayload,
    ) {
      if (isDisposed) return;

      setIsConnected(true);
      clearSocketError();
      scheduleJoinCharacterRoom();
    }

    function handleJoined(payload: AutoCombatJoinedPayload) {
      if (isDisposed) return;

      if (payload.characterId !== characterId) {
        return;
      }

      hasJoinedCharacterRoom = true;
      isJoiningCharacterRoom = false;

      setIsJoined(true);
      clearSocketError();
    }

    function handleLeft(payload: AutoCombatLeftPayload) {
      if (isDisposed) return;

      if (payload.characterId !== characterId) {
        return;
      }

      hasJoinedCharacterRoom = false;
      isJoiningCharacterRoom = false;

      setIsJoined(false);
    }

    function handleDisconnect() {
      if (isDisposed) return;

      hasJoinedCharacterRoom = false;
      isJoiningCharacterRoom = false;

      setIsConnected(false);
      setIsJoined(false);
    }

    function handleConnectError(error: Error) {
      if (isDisposed) return;

      hasJoinedCharacterRoom = false;
      isJoiningCharacterRoom = false;

      const message =
        error?.message ||
        'Não foi possível conectar ao WebSocket do combate.';

      setIsConnected(false);
      setIsJoined(false);
      safeSetError(message);
    }

    function handleSocketError(payload: AutoCombatSocketError) {
      if (isDisposed) return;

      hasJoinedCharacterRoom = false;
      isJoiningCharacterRoom = false;

      const message = payload.message || 'Erro no WebSocket do combate.';

      setIsJoined(false);
      safeSetError(message);
    }

    function handleStatus(payload: AutoCombatStatusResponse) {
      if (isDisposed) return;

      clearSocketError();
      handlersRef.current.onStatus?.(payload);
    }

      if (isDisposed) return;

      clearSocketError();
      handlersRef.current.onSessionUpdated?.(payload);

      if (isDisposed) return;

      clearSocketError();
      handlersRef.current.onFinished?.(payload);

      if (isDisposed) return;

      clearSocketError();
      handlersRef.current.onStopped?.(payload);
    }

      payload: AutoCombatRealtimeEvent,

      if (isDisposed) return;

      setLastEvent(payload);
      callback?.(payload);

    function handleMobSpawned(payload: AutoCombatRealtimeEvent) {
      handleRealtimeEvent(payload, handlersRef.current.onMobSpawned);
    }

    function handleHit(payload: AutoCombatRealtimeEvent) {
      handleRealtimeEvent(payload, handlersRef.current.onHit);
    }

    function handleMobDefeated(payload: AutoCombatRealtimeEvent) {
      handleRealtimeEvent(payload, handlersRef.current.onMobDefeated);
    }

    function handlePlayerDefeated(payload: AutoCombatRealtimeEvent) {
      handleRealtimeEvent(payload, handlersRef.current.onPlayerDefeated);
    }

    function handlePotionUsed(payload: AutoCombatRealtimeEvent) {
      handleRealtimeEvent(payload, handlersRef.current.onPotionUsed);
    }

    currentSocket.on('connect', handleConnect);
    currentSocket.on('disconnect', handleDisconnect);
    currentSocket.on('connect_error', handleConnectError);

    currentSocket.on('auto-combat:connected', handleAuthenticatedConnection);
    currentSocket.on('auto-combat:joined', handleJoined);
    currentSocket.on('auto-combat:left', handleLeft);
    currentSocket.on('auto-combat:error', handleSocketError);

    currentSocket.on('auto-combat:status', handleStatus);
    currentSocket.on('auto-combat:session-updated', handleSessionUpdated);
    currentSocket.on('auto-combat:finished', handleFinished);
    currentSocket.on('auto-combat:stopped', handleStopped);

    currentSocket.on('auto-combat:mob-spawned', handleMobSpawned);
    currentSocket.on('auto-combat:hit', handleHit);
    currentSocket.on('auto-combat:mob-defeated', handleMobDefeated);
    currentSocket.on('auto-combat:player-defeated', handlePlayerDefeated);
    currentSocket.on('auto-combat:potion-used', handlePotionUsed);

    if (currentSocket.connected) {
      setIsConnected(true);
      scheduleJoinCharacterRoom();
    }

    return () => {
      isDisposed = true;

      if (joinTimeoutId !== undefined) {
        window.clearTimeout(joinTimeoutId);
      }

      if (currentSocket.connected && characterId) {
        currentSocket.emit('auto-combat:leave', {
          characterId,
        });
      }

      currentSocket.off('connect', handleConnect);
      currentSocket.off('disconnect', handleDisconnect);
      currentSocket.off('connect_error', handleConnectError);

      currentSocket.off(
        'auto-combat:connected',
        handleAuthenticatedConnection,
      );
      currentSocket.off('auto-combat:joined', handleJoined);
      currentSocket.off('auto-combat:left', handleLeft);
      currentSocket.off('auto-combat:error', handleSocketError);

      currentSocket.off('auto-combat:status', handleStatus);
      currentSocket.off('auto-combat:session-updated', handleSessionUpdated);
      currentSocket.off('auto-combat:finished', handleFinished);
      currentSocket.off('auto-combat:stopped', handleStopped);

      currentSocket.off('auto-combat:hit', handleHit);
      currentSocket.off('auto-combat:mob-defeated', handleMobDefeated);
      currentSocket.off('auto-combat:player-defeated', handlePlayerDefeated);
      currentSocket.off('auto-combat:potion-used', handlePotionUsed);
    };
  }, [characterId, enabled]);

  return {
    socket,
    isConnected,
    isJoined,
    errorMessage,
    lastEvent,
  };
